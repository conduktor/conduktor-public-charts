DEPRECATION NOTICE: gateway.portRange is deprecated and will be removed in a future
  major chart version. Migrate to named listener mode using this starting-point config:

      listeners: true
    securityMode: "{{ $effectiveSecurityMode }}"
    listeners:
      internal:
        securityProtocol: {{ $legacyProtocol }}
        routing: {{ $legacyRouting }}
        ports:
          - "{{ $portSpec }}"
